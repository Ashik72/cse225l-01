void run_part_2() { 
 
    int count; 
 
    cin >> count; 
 
    dynArr2 obj2(count); 
 
    for (int i = 0; i < count; i++) { 
 
        int temp; 
 
        cin >> temp; 
 
        obj2.set(i, temp); 
 
    } 
 
    for (int i = 0; i < count; i++) { 
 
        cout << obj2.get(i) << endl; 
 
    } 
 
    cout << "Allocate: " << endl; 
 
    int all1; 
 
    cin >> all1; 
 
    obj2.allocate(all1); 
 
    for (int i = 0; i < all1; i++) { 
 
        int temp; 
 
        cin >> temp; 
 
        obj2.set(i, temp); 
 
    } 
 
    for (int i = 0; i < all1; i++) { 
 
        cout << obj2.get(i) << endl; 
 
    } 
 
    //obj2.~dynArr2(); 
 
 
} 
 
void run_part_3() {
